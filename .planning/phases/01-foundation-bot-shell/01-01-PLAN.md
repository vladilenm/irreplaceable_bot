2. Create src/config.ts:
```typescript
import 'dotenv/config';
import type { BotConfig } from './types/index.js';

function requireEnv(name: string): string {
  const value = process.env[name];
  if (!value) {
    throw new Error(`Missing required environment variable: ${name}`);
  }
  return value;
}

export const config: BotConfig = {
  botToken: requireEnv('BOT_TOKEN'),
  targetChatId: requireEnv('TARGET_CHAT_ID'),
  aiRadarThreadId: requireEnv('AI_RADAR_THREAD_ID'),
  digestCron: process.env['DIGEST_CRON'] ?? '0 6 * * *',
  aiApiKey: requireEnv('AI_API_KEY'),
  aiModel: process.env['AI_MODEL'] ?? 'claude-sonnet-4-20250514',
  logLevel: process.env['LOG_LEVEL'] ?? 'info',
  nodeEnv: process.env['NODE_ENV'] ?? 'production',
};
```

IMPORTANT: Use bracket notation `process.env['KEY']` (not dot notation) to satisfy noUncheckedIndexedAccess. The requireEnv helper throws on missing required vars. Optional vars use `??` with defaults.

SECURITY: BOT_TOKEN and AI_API_KEY are secrets. They are loaded from .env (gitignored) and NEVER logged. The config object is exported but consumers must not log it wholesale.

3. Create src/utils/logger.ts:
```typescript
import pino from 'pino';
import { config } from '../config.js';

export const logger = pino({
  level: config.logLevel,
  transport:
    config.nodeEnv === 'development'
      ? { target: 'pino-pretty', options: { colorize: true } }
      : undefined,
});
```

In production (NODE_ENV=production), pino outputs raw JSON (structured logs). In development, pino-pretty formats for readability.
